from PyQt5.QtWidgets import QApplication
from PyQt5.QtWebEngineWidgets import QWebEngineView
from PyQt5.QtCore import QTimer
import sys

def capture_screenshot(url, output_path, width=480, height=800): # test these resolution settings
    app = QApplication(sys.argv)

    # Set up a QWebEngineView for rendering
    web = QWebEngineView()
    web.resize(480, 800) # test these resolution settings
    web.load(url)

    # Capture the screenshot after the page loads
    def capture():
        web.grab().save(output_path)
        print(f"Screenshot saved to {output_path}")
        app.quit()

    # Wait for the page to load before capturing
    QTimer.singleShot(5000, capture)  # Adjust time for slow-loading pages
    web.show()
    sys.exit(app.exec_())

# Example usage
if __name__ == "__main__":
    capture_screenshot("https://www.gradyp.com/chamboard/index3.html", "webkit.png")
